307  Sample parameters = params[0];

309  try {

312  // Insert the sample data

316  // Release access

318  Log.d("SampleTask",

320  } catch (DataBaseException e) {

322  } catch (CollectMeasureException e) {
323  Log.e("SampleTask", "CollectMeasureException", e);
324  } catch (InterruptedException e) {
325  Log.e("SampleTaks", "InterruptedException", e);
326  }
327  return null;
328  }

330  @Override
331  protected void onPostExecute(Void result) {
332  // The task is over
333  sampleTaskCount--;
334  // Should we continue the recording ?
335  if (isRecording) {
336  // Get the elapsed time
337  long elapsedTime = System.currentTimeMillis() - startTime;
338  // Compensate the time taken for the insertion
339  if (elapsedTime > sampleRate) {
340  // Call again as soon as possible
341  RecordingHandler.this.sendEmptyMessage(MESSAGE_SAMPLE);
342  } else {
343  // We are on schedule, call again for the next sample
344  RecordingHandler.this.sendEmptyMessageDelayed(
345  MESSAGE_SAMPLE, sampleRate - elapsedTime);
346  }
347  // We don't modify the sample task count because we are
348  // requesting another one
349  } else {
350  // Finish recording
351  onFinishRecording();
352  }
353  }
