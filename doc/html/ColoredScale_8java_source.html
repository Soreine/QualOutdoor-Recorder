399  private int compatibleResolveSizeAndState(int size, int measureSpec,
400  int childMeasuredState) {
401  int result = size;
402  int specMode = MeasureSpec.getMode(measureSpec);
403  int specSize = MeasureSpec.getSize(measureSpec);
404  switch (specMode) {
405  case MeasureSpec.UNSPECIFIED:
406  result = size;
407  break;
408  case MeasureSpec.AT_MOST:
409  if (specSize < size) {
410  result = specSize | MEASURED_STATE_TOO_SMALL;
411  } else {
412  result = size;
413  }
414  break;
415  case MeasureSpec.EXACTLY:
416  result = specSize;
417  break;
418  }
419  return result | (childMeasuredState & MEASURED_STATE_MASK);
420  }
