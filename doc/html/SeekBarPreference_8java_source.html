166  public void onProgressChanged(SeekBar seekBar, int progress,
167  boolean fromUser) {
168  int newValue = progress + mMinValue;
169 
170  if (newValue > mMaxValue)
171  newValue = mMaxValue;
172  else if (newValue < mMinValue)
173  newValue = mMinValue;
174  else if (mInterval != 1 && newValue % mInterval != 0)
175  newValue = Math.round(((float) newValue) / mInterval) * mInterval;
176 
177  // change rejected, revert to the previous value
178  if (!callChangeListener(newValue)) {
179  seekBar.setProgress(mCurrentValue - mMinValue);
180  return;
